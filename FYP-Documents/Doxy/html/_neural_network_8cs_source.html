199 public void BackPropagate(float[] inputs, float[] expected)//backpropogation;
200 {
201 float[] output = FeedForward(inputs);//runs feed forward to ensure neurons are populated correctly
202
203 cost = 0;
204 for (int i = 0; i < output.Length; i++) cost += (float)Math.Pow(output[i] - expected[i], 2);//calculated cost of network
205 cost = cost / 2;//this value is not used in calculions, rather used to identify the performance of the network
206
207 float[][] gamma;
208
209
210 List<float[]> gammaList = new List<float[]>();
211 for (int i = 0; i < layers.Length; i++)
212 {
213 gammaList.Add(new float[layers[i]]);
214 }
215 gamma = gammaList.ToArray();//gamma initialization
216
217 int layer = layers.Length - 2;
218 for (int i = 0; i < output.Length; i++) gamma[layers.Length - 1][i] = (output[i] - expected[i]) * activateDer(output[i], layer);//Gamma calculation
219 for (int i = 0; i < layers[layers.Length - 1]; i++)//calculates the w' and b' for the last layer in the network
220 {
221 biases[layers.Length - 2][i] -= gamma[layers.Length - 1][i] * learningRate;
222 for (int j = 0; j < layers[layers.Length - 2]; j++)
223 {
224
225 weights[layers.Length - 2][i][j] -= gamma[layers.Length - 1][i] * neurons[layers.Length - 2][j] * learningRate;//*learning
226 }
227 }
228
229 for (int i = layers.Length - 2; i > 0; i--)//runs on all hidden layers
230 {
231 layer = i - 1;
232 for (int j = 0; j < layers[i]; j++)//outputs
233 {
234 gamma[i][j] = 0;
235 for (int k = 0; k < gamma[i + 1].Length; k++)
236 {
237 gamma[i][j] += gamma[i + 1][k] * weights[i][k][j];
238 }
239 gamma[i][j] *= activateDer(neurons[i][j], layer);//calculate gamma
240 }
241 for (int j = 0; j < layers[i]; j++)//itterate over outputs of layer
242 {
243 biases[i - 1][j] -= gamma[i][j] * learningRate;//modify biases of network
244 for (int k = 0; k < layers[i - 1]; k++)//itterate over inputs to layer
245 {
246 weights[i - 1][j][k] -= gamma[i][j] * neurons[i - 1][k] * learningRate;//modify weights of network
