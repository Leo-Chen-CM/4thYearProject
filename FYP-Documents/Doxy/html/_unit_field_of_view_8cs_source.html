155
156 for (int i = 0; i <= stepCount; i++)

162 List<Vector3> viewPoints = new List<Vector3>();
163 ViewCastInfo oldViewCast = new ViewCastInfo();

165 {
166 float angle = transform.eulerAngles.z - m_angle / 2 + stepAngleSize * i;
167 ViewCastInfo newViewCast = ViewCast(angle);
168 viewPoints.Add(newViewCast.point);

172 if (oldViewCast.hit != newViewCast.hit || (oldViewCast.hit && newViewCast.hit && edgeDstThresholdExceeded))
173 {
174 EdgeInfo edge = FindEdge(oldViewCast, newViewCast);
175 if (edge.pointA != Vector3.zero)

177 viewPoints.Add(edge.pointA);
178 }
179 if (edge.pointB != Vector3.zero)

181 viewPoints.Add(edge.pointB);
182 }
183 }
184 }
185 oldViewCast = newViewCast;
186 }

212 EdgeInfo FindEdge(ViewCastInfo minViewCast, ViewCastInfo maxViewCast)
213 {
214 float minAngle = minViewCast.angle;
215 float maxAngle = maxViewCast.angle;
216 Vector3 minPoint = Vector3.zero;
217 Vector3 maxPoint = Vector3.zero;
218
219 for (int i = 0; i < edgeResolveIterations; i++)
220 {
221 float angle = (minAngle + maxAngle) / 2;
222 ViewCastInfo newViewCast = ViewCast(angle);
223
224 bool edgeDstThresholdExceeded = Mathf.Abs(minViewCast.dst - newViewCast.dst) > edgeDstThreshold;
225 if (newViewCast.hit == minViewCast.hit && !edgeDstThresholdExceeded)
